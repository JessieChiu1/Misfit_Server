const S3 = require("aws-sdk").S3
require("dotenv").config()
 
const bucketName = process.env.AWS_BUCKET_NAME
const region = process.env.AWS_BUCKET_REGION
const accessKeyId = process.env.AWS_ACCESS_KEY
const secretAccessKey = process.env.AWS_SECRET_KEY
 
const s3 = new S3({
    region,
    accessKeyId,
    secretAccessKey
})
 
const uploadImage = async (file) => {
    const fileStream = fs.readFileSync(file.path)
    const mainKey = path.basename(file.originalname, path.extname(file.originalname)) + '-' + Date.now() + path.extname(file.originalname);

    const uploadParams = {
        Bucket: bucketName,
        Body: fileStream,
        Key: mainKey
    }

    const command = new PutObjectCommand(uploadParams);
    await s3.send(command);

    const location = `https://${bucketName}.s3.${region}.amazonaws.com/${mainKey}`

    return { 
        mainUrl: location,
        mainKey,
    }
    
}
 
const deleteImage = async (fileKey) => {
    const deleteParams = {
        Key: fileKey,
        Bucket: bucketName,
    }

    const command = new DeleteObjectCommand(deleteParams);
    await s3.send(command);
}
 
module.exports = {
    uploadImage,
    deleteImage
}
